User
	hasMany: Question, Answer, Comment, Notification, Badge, Privilege, Tag, Vote
	name
	real_name
	password
	email
	photo
	bio : Text
	birthday
	website
	location
	reputation
	created_at * a verifier
	last_login
	profile_views
	isAdmin

Privilege
	hasMany: User
	belongsTo: PrivilegeCategory
	name
	description
	reputation

PrivilegeCategory
	hasMany: Privilege
	name
	icon
Badge
	hasMany: User
	belongsTo: BadgeCategory
	name
	description

BadgeCategory
	hasMany: Badge
	name
	icon

Notification
	belongsTo: User
	type : {comment, answer, badge, privilege, ...}
	name (the name of the badge, privilege or question)
	url
	read: bool
	createdAt

Question
	belongsTo: User(author)
	hasMany: Tag, Answer, Vote, Comment
	hasOne: Answer(bestAnswer)
	title
	content: text
	createdAt
	modifiedAt
	hidden: bool
	views: int

Answer
	belongsTo: Question, User(author)
	hasMany: Vote, Comment
	content: text
	hidden: bool

Comment
	belongsTo: Answer, Question, User(author)
	hasMany: Vote
	content: text
	hidden: bool

Tag
	hasMany: Question, User
	name
	content: text

Vote
	belongsTo: User
	value: {-1, 1}
	createdAt

Page
	title
	content: text
